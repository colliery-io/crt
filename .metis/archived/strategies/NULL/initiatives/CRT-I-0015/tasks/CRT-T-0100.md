```rust
use crt_core::TestTerminal;

#[test]
fn test_cursor_home() {
    let mut t = TestTerminal::new(80, 24);
    t.send("Hello");
    t.send_escape("[H");
    t.assert_cursor_at(0, 0);
}

#[test]
fn test_cursor_absolute_position() {
    let mut t = TestTerminal::new(80, 24);
    t.send_escape("[10;20H");
    t.assert_cursor_at(19, 9); // 0-indexed
}

#[test]
fn test_cursor_up() {
    let mut t = TestTerminal::new(80, 24);
    t.send_escape("[5;5H"); // Start at row 5, col 5
    t.send_escape("[2A");   // Move up 2
    t.assert_cursor_at(4, 2);
}

#[test]
fn test_cursor_down() {
    let mut t = TestTerminal::new(80, 24);
    t.send_escape("[5;5H");
    t.send_escape("[3B");
    t.assert_cursor_at(4, 7);
}

#[test]
fn test_cursor_forward() {
    let mut t = TestTerminal::new(80, 24);
    t.send_escape("[1;1H");
    t.send_escape("[10C");
    t.assert_cursor_at(10, 0);
}

#[test]
fn test_cursor_backward() {
    let mut t = TestTerminal::new(80, 24);
    t.send_escape("[1;20H");
    t.send_escape("[5D");
    t.assert_cursor_at(14, 0);
}

#[test]
fn test_cursor_save_restore() {
    let mut t = TestTerminal::new(80, 24);
    t.send_escape("[5;10H");
    t.send_escape("7");      // Save cursor (DECSC)
    t.send_escape("[1;1H");  // Move elsewhere
    t.send_escape("8");      // Restore cursor (DECRC)
    t.assert_cursor_at(9, 4);
}

#[test]
fn test_cursor_column() {
    let mut t = TestTerminal::new(80, 24);
    t.send_escape("[5;1H");
    t.send_escape("[15G");   // Move to column 15
    t.assert_cursor_at(14, 4);
}
```

### File: `crt-core/tests/integration/text.rs`

```rust
use crt_core::TestTerminal;

#[test]
fn test_basic_text_output() {
    let mut t = TestTerminal::new(80, 24);
    t.send("Hello, World!");
    t.assert_line_contains(0, "Hello, World!");
}

#[test]
fn test_line_wrap() {
    let mut t = TestTerminal::new(10, 5);
    t.send("This is a long line that wraps");
    t.assert_line_contains(0, "This is a ");
    t.assert_line_contains(1, "long line ");
}

#[test]
fn test_overwrite() {
    let mut t = TestTerminal::new(80, 24);
    t.send("AAAA");
    t.send_escape("[1;1H");
    t.send("BB");
    t.assert_line_contains(0, "BBAA");
}

#[test]
fn test_newline() {
    let mut t = TestTerminal::new(80, 24);
    t.send("Line 1\r\nLine 2\r\nLine 3");
    t.assert_line_contains(0, "Line 1");
    t.assert_line_contains(1, "Line 2");
    t.assert_line_contains(2, "Line 3");
}

#[test]
fn test_tab() {
    let mut t = TestTerminal::new(80, 24);
    t.send("A\tB");
    // Default tab stop is 8
    t.assert_cell_is(0, 0, 'A');
    t.assert_cell_is(8, 0, 'B');
}
```

### File: `crt-core/tests/integration/erase.rs`

```rust
use crt_core::TestTerminal;

#[test]
fn test_erase_to_end_of_line() {
    let mut t = TestTerminal::new(80, 24);
    t.send("Hello World");
    t.send_escape("[1;6H");  // Move to 'W'
    t.send_escape("[K");     // Erase to end
    t.assert_line_contains(0, "Hello");
}

#[test]
fn test_erase_to_start_of_line() {
    let mut t = TestTerminal::new(80, 24);
    t.send("Hello World");
    t.send_escape("[1;6H");
    t.send_escape("[1K");
    // First 6 chars erased, rest remains
    t.assert_line_contains(0, "World");
}

#[test]
fn test_erase_entire_line() {
    let mut t = TestTerminal::new(80, 24);
    t.send("Hello World");
    t.send_escape("[2K");
    assert_eq!(t.line_content(0), "");
}

#[test]
fn test_erase_screen() {
    let mut t = TestTerminal::new(80, 24);
    t.send("Line 1\r\nLine 2\r\nLine 3");
    t.send_escape("[2J");
    assert_eq!(t.screen_content().trim(), "");
}
```
