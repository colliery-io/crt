### File: `src/managers/event_handler.rs`

```rust
use winit::event::{WindowEvent, DeviceEvent};
use winit::window::WindowId;
use crate::input::Command;

pub struct EventHandler;

impl EventHandler {
    /// Handle a window event, returns commands to execute
    pub fn handle_window_event(
        window_id: WindowId,
        event: WindowEvent,
        state: &mut WindowState,
        config: &Config,
    ) -> Vec<Command> {
        match event {
            WindowEvent::KeyboardInput { event, .. } => {
                Self::handle_keyboard(event, state, config)
            }
            WindowEvent::MouseInput { button, state: elem_state, .. } => {
                Self::handle_mouse_button(button, elem_state, state)
            }
            WindowEvent::CursorMoved { position, .. } => {
                Self::handle_cursor_move(position, state)
            }
            WindowEvent::Resized(size) => {
                Self::handle_resize(size, state)
            }
            WindowEvent::Focused(focused) => {
                vec![if focused { Command::RequestRedraw } else { vec![] }]
            }
            // ... other events
            _ => vec![],
        }
    }

    fn handle_keyboard(
        event: KeyEvent,
        state: &mut WindowState,
        config: &Config,
    ) -> Vec<Command> {
        // Use input/keyboard.rs pure functions
        if keyboard::is_shortcut(&event.logical_key, state.modifiers) {
            keyboard::shortcut_to_command(&event.logical_key, state.modifiers)
                .into_iter()
                .collect()
        } else if let Some(bytes) = keyboard::key_to_terminal_bytes(
            &event.logical_key,
            state.modifiers,
            state.terminal_mode(),
        ) {
            vec![Command::SendToPty(bytes)]
        } else {
            vec![]
        }
    }
    
    // ... other handlers
}
```

### Refactored `main.rs`

```rust
use crate::managers::{WindowManager, ConfigManager, EventHandler};

struct App {
    windows: WindowManager,
    config: ConfigManager,
    shared_gpu: Option<SharedGpuState>,
    modifiers: Modifiers,
}

impl ApplicationHandler for App {
    fn window_event(&mut self, event_loop: &ActiveEventLoop, id: WindowId, event: WindowEvent) {
        if let Some(state) = self.windows.get_mut(id) {
            let commands = EventHandler::handle_window_event(
                id, event, state, self.config.config()
            );
            self.execute_commands(commands);
        }
    }

    fn about_to_wait(&mut self, event_loop: &ActiveEventLoop) {
        // Check for config changes
        if let Some(change) = self.config.poll_changes() {
            self.handle_config_change(change);
        }
        
        // Render all windows
        for (_, state) in self.windows.iter_mut() {
            if state.needs_render() {
                render_frame(state, self.config.theme());
            }
        }
    }
}

fn main() {
    let event_loop = EventLoop::new().unwrap();
    let mut app = App::new().unwrap();
    event_loop.run_app(&mut app).unwrap();
}
```
