### File: `crt-core/src/fake_pty.rs`

```rust
use std::collections::VecDeque;
use std::io::{self, Read, Write};

/// A fake PTY for testing that doesn't spawn real processes
pub struct FakePty {
    /// Bytes queued to be "received" from the PTY
    output_queue: VecDeque<u8>,
    /// Bytes that were "sent" to the PTY
    input_buffer: Vec<u8>,
    /// Whether the PTY is "alive"
    alive: bool,
}

impl FakePty {
    pub fn new() -> Self {
        Self {
            output_queue: VecDeque::new(),
            input_buffer: Vec::new(),
            alive: true,
        }
    }

    /// Queue bytes to be received by the terminal
    pub fn queue_output(&mut self, data: &[u8]) {
        self.output_queue.extend(data);
    }

    /// Queue a string to be received
    pub fn queue_str(&mut self, s: &str) {
        self.queue_output(s.as_bytes());
    }

    /// Queue an ANSI escape sequence
    pub fn queue_escape(&mut self, sequence: &str) {
        self.queue_output(format!("\x1b{}", sequence).as_bytes());
    }

    /// Queue cursor movement: ESC[row;colH
    pub fn queue_cursor_move(&mut self, row: usize, col: usize) {
        self.queue_escape(&format!("[{};{}H", row, col));
    }

    /// Queue clear screen: ESC[2J
    pub fn queue_clear_screen(&mut self) {
        self.queue_escape("[2J");
    }

    /// Queue set color: ESC[38;2;r;g;bm (24-bit foreground)
    pub fn queue_fg_color(&mut self, r: u8, g: u8, b: u8) {
        self.queue_escape(&format!("[38;2;{};{};{}m", r, g, b));
    }

    /// Queue reset attributes: ESC[0m
    pub fn queue_reset(&mut self) {
        self.queue_escape("[0m");
    }

    /// Get bytes that were written to the PTY (terminal input)
    pub fn read_input(&mut self) -> Vec<u8> {
        std::mem::take(&mut self.input_buffer)
    }

    /// Get input as string (lossy conversion)
    pub fn read_input_string(&mut self) -> String {
        String::from_utf8_lossy(&self.read_input()).into_owned()
    }

    /// Check if any output is pending
    pub fn has_pending_output(&self) -> bool {
        !self.output_queue.is_empty()
    }

    /// Simulate PTY death
    pub fn kill(&mut self) {
        self.alive = false;
    }

    /// Check if PTY is alive
    pub fn is_alive(&self) -> bool {
        self.alive
    }
}

impl Read for FakePty {
    fn read(&mut self, buf: &mut [u8]) -> io::Result<usize> {
        if !self.alive {
            return Ok(0);
        }
        
        let count = buf.len().min(self.output_queue.len());
        for i in 0..count {
            buf[i] = self.output_queue.pop_front().unwrap();
        }
        Ok(count)
    }
}

impl Write for FakePty {
    fn write(&mut self, buf: &[u8]) -> io::Result<usize> {
        if !self.alive {
            return Err(io::Error::new(io::ErrorKind::BrokenPipe, "PTY dead"));
        }
        self.input_buffer.extend_from_slice(buf);
        Ok(buf.len())
    }

    fn flush(&mut self) -> io::Result<()> {
        Ok(())
    }
}

impl Default for FakePty {
    fn default() -> Self {
        Self::new()
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn test_queue_and_read() {
        let mut pty = FakePty::new();
        pty.queue_str("Hello");
        
        let mut buf = [0u8; 10];
        let n = pty.read(&mut buf).unwrap();
        
        assert_eq!(n, 5);
        assert_eq!(&buf[..n], b"Hello");
    }

    #[test]
    fn test_write_and_read_input() {
        let mut pty = FakePty::new();
        pty.write_all(b"test input").unwrap();
        
        assert_eq!(pty.read_input_string(), "test input");
    }

    #[test]
    fn test_escape_sequences() {
        let mut pty = FakePty::new();
        pty.queue_cursor_move(5, 10);
        
        let mut buf = [0u8; 20];
        let n = pty.read(&mut buf).unwrap();
        
        assert_eq!(&buf[..n], b"\x1b[5;10H");
    }
}
```
