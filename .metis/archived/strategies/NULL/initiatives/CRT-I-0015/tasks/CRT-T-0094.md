## Implementation Notes

### File: `crt-renderer/src/traits.rs`

```rust
use alacritty_terminal::index::Point;

/// Color as RGBA floats (GPU-agnostic)
#[derive(Debug, Clone, Copy)]
pub struct Color {
    pub r: f32,
    pub g: f32,
    pub b: f32,
    pub a: f32,
}

/// A single cell to render
#[derive(Debug, Clone)]
pub struct CellContent {
    pub character: char,
    pub position: Point,
    pub fg_color: Color,
    pub bg_color: Option<Color>,
    pub bold: bool,
    pub italic: bool,
    pub underline: bool,
}

/// Cursor rendering info
#[derive(Debug, Clone)]
pub struct CursorInfo {
    pub position: Point,
    pub style: CursorStyle,
    pub color: Color,
    pub visible: bool,
}

#[derive(Debug, Clone, Copy)]
pub enum CursorStyle {
    Block,
    Underline,
    Beam,
}

/// Selection range for highlighting
#[derive(Debug, Clone)]
pub struct SelectionRange {
    pub start: Point,
    pub end: Point,
    pub color: Color,
}

/// Tab info for rendering
#[derive(Debug, Clone)]
pub struct TabRenderInfo {
    pub title: String,
    pub active: bool,
    pub has_activity: bool,
    pub bounds: Rect,
}

#[derive(Debug, Clone, Copy)]
pub struct Rect {
    pub x: f32,
    pub y: f32,
    pub width: f32,
    pub height: f32,
}

/// Trait for rendering terminal text content
pub trait TextRenderer {
    /// Render a grid of cells
    fn render_cells(&mut self, cells: &[CellContent]);
    
    /// Render the cursor
    fn render_cursor(&mut self, cursor: &CursorInfo);
    
    /// Render selection highlighting
    fn render_selection(&mut self, ranges: &[SelectionRange]);
    
    /// Clear the text buffer
    fn clear(&mut self);
}

/// Trait for rendering UI elements
pub trait UiRenderer {
    /// Render the tab bar
    fn render_tabs(&mut self, tabs: &[TabRenderInfo], bar_bounds: Rect);
    
    /// Render search match highlights
    fn render_search_matches(
        &mut self,
        matches: &[Rect],
        current_index: usize,
        highlight_color: Color,
        current_color: Color,
    );
    
    /// Render context menu
    fn render_context_menu(
        &mut self,
        position: (f32, f32),
        items: &[&str],
        selected: usize,
    );
    
    /// Render bell flash overlay
    fn render_bell_flash(&mut self, intensity: f32);
}

/// Trait for rendering backdrop effects
pub trait EffectsRenderer {
    /// Update effect animations
    fn update(&mut self, dt: f32);
    
    /// Render all active effects
    fn render(&mut self);
    
    /// Check if any effects need continuous rendering
    fn needs_animation(&self) -> bool;
}
```

### Integration with Existing Code

The existing `GridRenderer`, `RectRenderer`, `TerminalVelloRenderer` can implement these traits, wrapping their GPU-specific internals.

### Dependencies

None - defines interfaces only.

### Notes

- Traits use value types (Color, Point, Rect) not GPU handles
- This enables MockRenderer to record calls without GPU
- Existing code continues to work, traits are additive
