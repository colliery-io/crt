/// Metrics for terminal cell layout
#[derive(Debug, Clone, Copy)]
pub struct CellMetrics {
    /// Width of a single character cell in pixels
    pub cell_width: f32,
    /// Height of a single character cell in pixels  
    pub cell_height: f32,
    /// Padding around the terminal content
    pub padding: Padding,
    /// Height of the tab bar (0 if hidden)
    pub tab_bar_height: f32,
}

#[derive(Debug, Clone, Copy, Default)]
pub struct Padding {
    pub top: f32,
    pub right: f32,
    pub bottom: f32,
    pub left: f32,
}

impl Padding {
    pub fn uniform(value: f32) -> Self {
        Self {
            top: value,
            right: value,
            bottom: value,
            left: value,
        }
    }
}

/// Window dimensions
#[derive(Debug, Clone, Copy)]
pub struct WindowSize {
    pub width: u32,
    pub height: u32,
    pub scale_factor: f32,
}

/// Terminal grid dimensions
#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub struct TerminalSize {
    pub cols: usize,
    pub rows: usize,
}

impl CellMetrics {
    pub fn new(cell_width: f32, cell_height: f32, padding: Padding, tab_bar_height: f32) -> Self {
        Self {
            cell_width,
            cell_height,
            padding,
            tab_bar_height,
        }
    }

    /// Calculate the content area available for the terminal grid
    pub fn content_area(&self, window: WindowSize) -> ContentArea {
        let physical_width = window.width as f32;
        let physical_height = window.height as f32;
        
        ContentArea {
            x: self.padding.left,
            y: self.padding.top + self.tab_bar_height,
            width: (physical_width - self.padding.left - self.padding.right).max(0.0),
            height: (physical_height - self.padding.top - self.padding.bottom - self.tab_bar_height).max(0.0),
        }
    }

    /// Calculate terminal dimensions (cols, rows) from window size
    pub fn calculate_terminal_size(&self, window: WindowSize) -> TerminalSize {
        let content = self.content_area(window);
        
        let cols = (content.width / self.cell_width).floor() as usize;
        let rows = (content.height / self.cell_height).floor() as usize;
        
        TerminalSize {
            cols: cols.max(1),
            rows: rows.max(1),
        }
    }

    /// Calculate window size needed for given terminal dimensions
    pub fn window_size_for_terminal(&self, size: TerminalSize) -> (f32, f32) {
        let content_width = size.cols as f32 * self.cell_width;
        let content_height = size.rows as f32 * self.cell_height;
        
        let width = content_width + self.padding.left + self.padding.right;
        let height = content_height + self.padding.top + self.padding.bottom + self.tab_bar_height;
        
        (width, height)
    }
}

#[derive(Debug, Clone, Copy)]
pub struct ContentArea {
    pub x: f32,
    pub y: f32,
    pub width: f32,
    pub height: f32,
}

#[cfg(test)]
mod tests {
    use super::*;

    fn test_metrics() -> CellMetrics {
        CellMetrics::new(10.0, 20.0, Padding::uniform(5.0), 30.0)
    }

    #[test]
    fn test_content_area() {
        let metrics = test_metrics();
        let window = WindowSize { width: 800, height: 600, scale_factor: 1.0 };
        
        let area = metrics.content_area(window);
        
        assert_eq!(area.x, 5.0);
        assert_eq!(area.y, 35.0); // padding + tab bar
        assert_eq!(area.width, 790.0); // 800 - 5 - 5
        assert_eq!(area.height, 560.0); // 600 - 5 - 5 - 30
    }

    #[test]
    fn test_terminal_size_calculation() {
        let metrics = test_metrics();
        let window = WindowSize { width: 800, height: 600, scale_factor: 1.0 };
        
        let size = metrics.calculate_terminal_size(window);
        
        assert_eq!(size.cols, 79); // 790 / 10 = 79
        assert_eq!(size.rows, 28); // 560 / 20 = 28
    }

    #[test]
    fn test_minimum_size() {
        let metrics = test_metrics();
        let window = WindowSize { width: 10, height: 10, scale_factor: 1.0 };
        
        let size = metrics.calculate_terminal_size(window);
        
        assert_eq!(size.cols, 1);
        assert_eq!(size.rows, 1);
    }

    #[test]
    fn test_window_size_for_terminal() {
        let metrics = test_metrics();
        let term_size = TerminalSize { cols: 80, rows: 24 };
        
        let (width, height) = metrics.window_size_for_terminal(term_size);
        
        assert_eq!(width, 810.0); // 80*10 + 5 + 5
        assert_eq!(height, 520.0); // 24*20 + 5 + 5 + 30
    }
}
