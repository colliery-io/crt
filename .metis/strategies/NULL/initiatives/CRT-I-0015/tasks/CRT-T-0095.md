```rust
#[cfg(test)]
use super::traits::*;

#[derive(Debug, Clone)]
pub enum RenderCall {
    // Text rendering
    Cells(Vec<CellContent>),
    Cursor(CursorInfo),
    Selection(Vec<SelectionRange>),
    ClearText,
    
    // UI rendering
    Tabs { tabs: Vec<TabRenderInfo>, bounds: Rect },
    SearchMatches { matches: Vec<Rect>, current: usize },
    ContextMenu { position: (f32, f32), items: Vec<String>, selected: usize },
    BellFlash(f32),
    
    // Effects
    EffectsUpdate(f32),
    EffectsRender,
}

#[derive(Default)]
pub struct MockRenderer {
    pub calls: Vec<RenderCall>,
    pub needs_animation: bool,
}

impl MockRenderer {
    pub fn new() -> Self {
        Self::default()
    }

    pub fn clear_calls(&mut self) {
        self.calls.clear();
    }

    // Assertion helpers
    
    pub fn assert_cell_rendered(&self, char: char, col: usize, line: i32) -> bool {
        self.calls.iter().any(|call| {
            if let RenderCall::Cells(cells) = call {
                cells.iter().any(|c| {
                    c.character == char && 
                    c.position.column.0 == col && 
                    c.position.line.0 == line
                })
            } else {
                false
            }
        })
    }

    pub fn assert_cursor_at(&self, col: usize, line: i32) -> bool {
        self.calls.iter().any(|call| {
            if let RenderCall::Cursor(cursor) = call {
                cursor.position.column.0 == col && cursor.position.line.0 == line
            } else {
                false
            }
        })
    }

    pub fn assert_selection_exists(&self) -> bool {
        self.calls.iter().any(|call| {
            matches!(call, RenderCall::Selection(ranges) if !ranges.is_empty())
        })
    }

    pub fn cell_count(&self) -> usize {
        self.calls.iter()
            .filter_map(|call| {
                if let RenderCall::Cells(cells) = call {
                    Some(cells.len())
                } else {
                    None
                }
            })
            .sum()
    }
}

impl TextRenderer for MockRenderer {
    fn render_cells(&mut self, cells: &[CellContent]) {
        self.calls.push(RenderCall::Cells(cells.to_vec()));
    }

    fn render_cursor(&mut self, cursor: &CursorInfo) {
        self.calls.push(RenderCall::Cursor(cursor.clone()));
    }

    fn render_selection(&mut self, ranges: &[SelectionRange]) {
        self.calls.push(RenderCall::Selection(ranges.to_vec()));
    }

    fn clear(&mut self) {
        self.calls.push(RenderCall::ClearText);
    }
}

impl UiRenderer for MockRenderer {
    fn render_tabs(&mut self, tabs: &[TabRenderInfo], bar_bounds: Rect) {
        self.calls.push(RenderCall::Tabs {
            tabs: tabs.to_vec(),
            bounds: bar_bounds,
        });
    }

    fn render_search_matches(
        &mut self,
        matches: &[Rect],
        current_index: usize,
        _highlight_color: Color,
        _current_color: Color,
    ) {
        self.calls.push(RenderCall::SearchMatches {
            matches: matches.to_vec(),
            current: current_index,
        });
    }

    fn render_context_menu(
        &mut self,
        position: (f32, f32),
        items: &[&str],
        selected: usize,
    ) {
        self.calls.push(RenderCall::ContextMenu {
            position,
            items: items.iter().map(|s| s.to_string()).collect(),
            selected,
        });
    }

    fn render_bell_flash(&mut self, intensity: f32) {
        self.calls.push(RenderCall::BellFlash(intensity));
    }
}

impl EffectsRenderer for MockRenderer {
    fn update(&mut self, dt: f32) {
        self.calls.push(RenderCall::EffectsUpdate(dt));
    }

    fn render(&mut self) {
        self.calls.push(RenderCall::EffectsRender);
    }

    fn needs_animation(&self) -> bool {
        self.needs_animation
    }
}

#[cfg(test)]
mod tests {
    use super::*;
    use alacritty_terminal::index::{Point, Column, Line};

    #[test]
    fn test_mock_records_cells() {
        let mut mock = MockRenderer::new();
        
        let cells = vec![CellContent {
            character: 'H',
            position: Point::new(Line(0), Column(0)),
            fg_color: Color { r: 1.0, g: 1.0, b: 1.0, a: 1.0 },
            bg_color: None,
            bold: false,
            italic: false,
            underline: false,
        }];
        
        mock.render_cells(&cells);
        
        assert!(mock.assert_cell_rendered('H', 0, 0));
        assert!(!mock.assert_cell_rendered('X', 0, 0));
    }

    #[test]
    fn test_mock_records_cursor() {
        let mut mock = MockRenderer::new();
        
        mock.render_cursor(&CursorInfo {
            position: Point::new(Line(5), Column(10)),
            style: CursorStyle::Block,
            color: Color { r: 1.0, g: 1.0, b: 1.0, a: 1.0 },
            visible: true,
        });
        
        assert!(mock.assert_cursor_at(10, 5));
    }
}
```
