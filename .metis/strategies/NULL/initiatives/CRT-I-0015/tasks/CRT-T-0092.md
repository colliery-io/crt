## Implementation Notes

### File: `src/managers/config_manager.rs`

```rust
use crate::config::Config;
use crate::watcher::ConfigWatcher;
use crt_theme::Theme;
use std::path::PathBuf;

pub enum ConfigChange {
    ConfigReloaded,
    ThemeReloaded,
    Error(String),
}

pub struct ConfigManager {
    config: Config,
    theme: Theme,
    watcher: Option<ConfigWatcher>,
    config_path: PathBuf,
    theme_path: PathBuf,
}

impl ConfigManager {
    pub fn new(config_path: PathBuf, theme_path: PathBuf) -> Result<Self, Error> {
        let config = Config::load(&config_path)?;
        let theme = Theme::load(&theme_path)?;
        
        Ok(Self {
            config,
            theme,
            watcher: None,
            config_path,
            theme_path,
        })
    }

    /// Start watching config files for changes
    pub fn start_watching(&mut self) -> Result<(), Error> {
        let watcher = ConfigWatcher::new(&self.config_path, &self.theme_path)?;
        self.watcher = Some(watcher);
        Ok(())
    }

    /// Check for file changes and reload if needed
    pub fn poll_changes(&mut self) -> Option<ConfigChange> {
        let watcher = self.watcher.as_mut()?;
        
        if watcher.config_changed() {
            match Config::load(&self.config_path) {
                Ok(config) => {
                    self.config = config;
                    Some(ConfigChange::ConfigReloaded)
                }
                Err(e) => Some(ConfigChange::Error(e.to_string())),
            }
        } else if watcher.theme_changed() {
            match Theme::load(&self.theme_path) {
                Ok(theme) => {
                    self.theme = theme;
                    Some(ConfigChange::ThemeReloaded)
                }
                Err(e) => Some(ConfigChange::Error(e.to_string())),
            }
        } else {
            None
        }
    }

    /// Force reload config
    pub fn reload_config(&mut self) -> Result<(), Error> {
        self.config = Config::load(&self.config_path)?;
        Ok(())
    }

    /// Force reload theme
    pub fn reload_theme(&mut self) -> Result<(), Error> {
        self.theme = Theme::load(&self.theme_path)?;
        Ok(())
    }

    /// Get current config
    pub fn config(&self) -> &Config {
        &self.config
    }

    /// Get current theme
    pub fn theme(&self) -> &Theme {
        &self.theme
    }
}
```
